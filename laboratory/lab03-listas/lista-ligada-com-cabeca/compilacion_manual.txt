# genera el archivo compilado app_main.o en la carpeta obj
gcc -I include -Wall -Werror -c apps/app_main.c -o obj/app_main.o

gcc obj/app_main.o obj/lista_enlazada_con_cabeza.o -o bin/app_main.out
